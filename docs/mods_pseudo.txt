BEGIN FindMAX(inArray)
	Let MaxIndex = 1
	Let Max = inArray[1]
	Let i = 1
	WHILE i < NumElements in inArray
	  IF inArray[i] > Max
	    Let Max = inArray[i]
		Let MaxIndex = i
	  ENDIF
	  
	  i = i + 1
	ENDWHILE
	
	Display “The highest value is ” Max “ at position ” MaxIndex

	return [Max, MaxIndex]
END FindMAX

BEGIN SumArrayContents(inputArray)
  Let i = 1
  Let total = 0
  WHILE i < NumElements in inputArray
    Let total = total + inputArray[i]
	i = i + 1
  ENDWHILE
  Display “ The sum of all of the elements in the array = ” total
  return total
END SumArrayContents
